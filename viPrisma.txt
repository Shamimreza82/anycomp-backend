enum Role {
  HR
  USER
  ADMIN
  MODERATOR
}

model User {
  id              String   @id @default(uuid())
  fullName        String?  @map("full_name")
  email           String   @unique
  phone           String?  @unique
  isActive        Boolean  @default(true) @map("is_active")
  isEmailVerified Boolean  @default(false) @map("is_email_verified")
  password        String
  role            Role     @default(USER)
  createdAt       DateTime @default(now()) @map("created_at")
  profile         Profile?

  certificates Certificate[]
  documents    Documents[]

  @@index([id, email])
  @@map("users")
}

////////////////////////// Profile //////////////////////////////

enum BloodGroup {
  A_POS
  A_NEG
  B_POS
  B_NEG
  O_POS
  O_NEG
  AB_POS
  AB_NEG
}

model Profile {
  id               String           @id @default(uuid())
  fatherName       String?          @map("father_name")
  motherName       String?          @map("mother_name")
  spouseName       String?          @map("spouse_name")
  avater           String?
  careerObjective  String?          @map("career_objective")
  dob              DateTime?
  height           String?
  weight           String?
  bloodGroup       BloodGroup?      @map("blood_group")
  nationality      String?
  religion         String?
  maritalStatus    Boolean          @default(false) @map("marital_status")
  isVerified       Boolean          @default(false) @map("is_verified")
  totalExperience  String?          @map("total_experience")
  portfolioLink    String?          @map("portfolio_link")
  presentAddress   Address?         @relation("PresentAddress")
  permanentAddress Address?         @relation("PermanentAddress")
  socialLinks      SocialLink[]
  skills           Skill[]
  workExperiences  WorkExperience[]
  educations       Education[]
  resume           String?
  certificates     Certificate[]
  createdAt        DateTime         @default(now()) @map("created_at")
  updatedAt        DateTime         @updatedAt @map("updated_at")
  user             User             @relation(fields: [userId], references: [id], onDelete: Cascade, onUpdate: Cascade)
  userId           String           @unique @map("user_id")
  documents        Documents[]

  @@index([id, userId])
  @@map("profiles")
}

model SocialLink {
  id        String  @id @default(uuid())
  url       String
  profile   Profile @relation(fields: [profileId], references: [id], onDelete: Cascade, onUpdate: Cascade)
  profileId String  @map("profile_id")

  @@map("social_links")
}

model Skill {
  id        String  @id @default(uuid())
  name      String
  profile   Profile @relation(fields: [profileId], references: [id], onDelete: Cascade, onUpdate: Cascade)
  profileId String  @map("profile_id")

  @@map("skills")
}

model Education {
  id        String    @id @default(uuid())
  degree    String
  academy   String
  startDate DateTime  @map("start_date")
  endDate   DateTime? @map("end_date")
  profile   Profile   @relation(fields: [profileId], references: [id], onDelete: Cascade, onUpdate: Cascade)
  profileId String    @map("profile_id")

  @@map("educations")
}

model Address {
  id String @id @default(uuid())

  address            String
  thana              String
  district           String
  country            String
  zip                String
  presentOf          Profile? @relation("PresentAddress", fields: [presentProfileId], references: [id], onDelete: Cascade, onUpdate: Cascade)
  presentProfileId   String?  @unique
  permanentOf        Profile? @relation("PermanentAddress", fields: [permanentProfileId], references: [id], onDelete: Cascade, onUpdate: Cascade)
  permanentProfileId String?  @unique

  @@map("addresses")
}

model WorkExperience {
  id String @id @default(uuid())

  jobTitle        String    @map("job_title")
  companyTitle    String    @map("company_title")
  companyLocation String    @map("company_location")
  jobRole         String    @map("job_role")
  startDate       DateTime  @map("start_date")
  endDate         DateTime? @map("end_date")
  profile         Profile   @relation(fields: [profileId], references: [id], onDelete: Cascade, onUpdate: Cascade)
  profileId       String    @map("profile_id")

  @@map("work_experiences")
}

//////////////////////// Profile ////////////////////

model Certificate {
  id         String   @id @default(uuid())
  profileId  String?  @map("profile_id")
  name       String // name of the certificate
  filePath   String   @map("file_path")
  uploadedAt DateTime @default(now()) @map("uploaded_at")
  profile    Profile? @relation(fields: [profileId], references: [id], onDelete: Cascade, onUpdate: Cascade)
  user       User?    @relation(fields: [userId], references: [id])
  userId     String?

  @@map("certificates")
}

//////////////////// Profile ////////////////////

model DocumentType {
  id   String @id @default(uuid())
  name String
}

model Documents {
  id         String   @id @default(uuid())
  name       String // name of the certificate
  filePath   String   @map("file_path")
  uploadedAt DateTime @default(now()) @map("uploaded_at")
}
